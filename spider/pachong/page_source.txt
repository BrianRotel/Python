                    var res = [],
                        len = str.length;
                    for (var i = 0; i &lt; len; ++i) {
                        res[i] = ("00" + str.charCodeAt(i).toString(16)).slice(-4);
                    }
                    return str ? "\\u" + res.join("\\u") : "";
                },
                parse: function (str) {
                    str = str.replace(/\\/g, "%");
                    return unescape(str);
                }
            },
            // 设置输入框文本

                if (search) {

                    arr1 = arr[1].split('&amp;')[0];
                    if (arr[0] == 'kw') {
                        result = this.Unicode.parse(arr1);

                                        &lt;i&gt;'+ (i+1) +'&lt;/i&gt;\

                                    &lt;/dl&gt;'
